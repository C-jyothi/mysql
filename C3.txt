-- 1. Insert the 3 books
INSERT INTO books (book_id, title, author, price, stock)
VALUES
(1, "Learn SQL", "John Smith", 400, 10),
(2, "Mastering Python", "Jane Doe", 600, 5),
(3, "HTML & CSS Basics", "Alan Webb", 300, 8);

-- 2. Increase the price of 'Learn SQL' by 50 and update stock to 12
UPDATE books
SET price = price + 50,
    stock = 12
WHERE title = "Learn SQL";

-- 3. Decrease stock by 2 for books with price greater than 500
UPDATE books
SET stock = stock - 2
WHERE price > 500;

-- 4. Delete the book with book_id = 3
DELETE FROM books
WHERE book_id = 3;
