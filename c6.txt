-- Show title and author from both tables
SELECT title, author
FROM books
UNION
SELECT title, author
FROM bestsellers;

-- Display books with price greater than 400
SELECT *
FROM books
WHERE price > 400;

-- Show the average price of books
SELECT AVG(price) AS average_price
FROM books;

-- Display the number of books
SELECT COUNT(*) AS total_books
FROM books;

-- Show title and author with aliases
SELECT title AS "Book Title",
       author AS "Written By"
FROM books;
