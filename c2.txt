-- 1. Show all unique product categories
SELECT DISTINCT category
FROM products;

-- 2. Select all products that are in stock and have a price less than 500
SELECT *
FROM products
WHERE in_stock = 'Yes' AND price < 500;

-- 3. Select all products that are not in stock or have a price greater than 1000
SELECT *
FROM products
WHERE in_stock = 'No' OR price > 1000;

-- 4. Show names and prices of all products, sorted by price from highest to lowest
SELECT name, price
FROM products
ORDER BY price DESC;

-- 5. Display name and price with 18% tax added
SELECT name,
       price * 1.18 AS price_with_tax
FROM products;
